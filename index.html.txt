        let currentIdx = 0;
        let score = 0;
        let userAnswers = []; 

        // 3. 퀴즈 렌더링 함수
        function renderQuestion() {
            const currentQuiz = quizData[currentIdx];
            
            document.getElementById('question-number').innerText = `Question ${currentIdx + 1} / ${quizData.length}`;
            const progressPercent = ((currentIdx + 1) / quizData.length) * 100;
            document.getElementById('progress-bar').style.width = `${progressPercent}%`;
            
            document.getElementById('question-text').innerText = currentQuiz.q;
            
            const optionsGroup = document.getElementById('options-group');
            optionsGroup.innerHTML = '';
            
            currentQuiz.options.forEach((opt) => {
                const button = document.createElement('button');
                button.className = 'option-btn';
                button.innerText = opt.text;
                button.onclick = () => handleSelectOption(opt);
                optionsGroup.appendChild(button);
            });
        }

        // 4. 보기 선택 처리
        function handleSelectOption(selectedOpt) {
            if (selectedOpt.isCorrect) {
                score++;
            }
            userAnswers.push(selectedOpt);

            if (currentIdx < quizData.length - 1) {
                currentIdx++;
                renderQuestion();
            } else {
                showResults();
            }
        }

        // 5. 결과 화면 출력 및 데이터베이스 전송
        function showResults() {
            document.getElementById('quiz-view').style.display = 'none';
            document.getElementById('result-view').style.display = 'block';
            document.getElementById('score-display').innerText = `${score} / ${quizData.length}`;

            const reviewList = document.getElementById('review-list');
            reviewList.innerHTML = '';

            quizData.forEach((quiz, i) => {
                const userAns = userAnswers[i];
                const correctAns = quiz.options.find(o => o.isCorrect);

                const card = document.createElement('div');
                card.className = 'review-card';

                card.innerHTML = `
                    <div class="review-q"><strong>Q${i+1}.</strong> ${quiz.q}</div>
                    <span class="review-status ${userAns.isCorrect ? 'status-correct' : 'status-incorrect'}">
                        ${userAns.isCorrect ? '정답' : '오답'}
                    </span>
                    <div class="review-ans">선택한 답변: <strong>${userAns.text}</strong></div>
                    ${!userAns.isCorrect ? `<div class="review-ans" style="color:var(--success-color)">올바른 정답: <strong>${correctAns.text}</strong></div>` : ''}
                    <div class="review-rationale">💡 <strong>해설:</strong> ${correctAns.rationale}</div>
                `;
                reviewList.appendChild(card);
            });

            // 데이터 전송
            sendDataToSupabase();
        }

        // 6. Supabase 실시간 적재 로직
        async function sendDataToSupabase() {
            if (!_supabase) return;
            
            try {
                // 수파베이스 테이블 구조에 안전하게 insert
                const { data, error } = await _supabase
                    .from('quiz_results') 
                    .insert([
                        { 
                            score: score, 
                            total_questions: quizData.length,
                            q1_answer: userAnswers[0]?.text,
                            q2_answer: userAnswers[1]?.text,
                            q3_answer: userAnswers[2]?.text,
                            created_at: new Date()
                        }
                    ]);
                
                if (error) throw error;
                console.log("데이터 백엔드 저장 완료");
            } catch (err) {
                console.error("저장 실패:", err.message);
            }
        }

        // 7. 다시하기 초기화
        function restartQuiz() {
            currentIdx = 0;
            score = 0;
            userAnswers = [];
            document.getElementById('result-view').style.display = 'none';
            document.getElementById('quiz-view').style.display = 'block';
            renderQuestion();
        }

        // 초기 시작
        window.onload = () => {
            renderQuestion();
        };
